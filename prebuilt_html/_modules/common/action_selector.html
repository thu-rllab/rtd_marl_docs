[docs]class LinearEpsilonDecay(EpsilonDecayBase):
    """
    The class for the linear epsilon decay strategy.

    Args:
        epsilon_start (float):
            The starting value of the epsilon.
        epsilon_end (float):
            The ending value of the epsilon.
        time_length (int):
            The time length of the epsilon decaying from ``epsilon_start`` to ``epsilon_end``.
    """
    def __init__(self, epsilon_start: float, epsilon_end: float, time_length: int):
        super(LinearEpsilonDecay, self).__init__(epsilon_start, epsilon_end, time_length)

        self.decay_type = 'linear'
        self.epsilon_delta = (self.epsilon_start - self.epsilon_end) / self.time_length


[docs]    def getEpsilon(self, t_step: int) -> torch.Tensor:
        """
        Obtain the epsilon for the given t step.

        Args:
            t_step (int or torch.Tensor):
                The t step or steps.

        Returns:
            torch.Tensor: **epsilon**: The epsilon for the given ``t_step(int)`` with shape [1, ] or ``t_step(torch.Tensor)`` with shape [batch_size, ].
        """
        if isinstance(t_step, int):
            epsilon = max(self.epsilon_end, self.epsilon_start - self.epsilon_delta * t_step)
            return torch.as_tensor([epsilon])
        elif isinstance(t_step, torch.Tensor):
            epsilon = self.epsilon_start - self.epsilon_delta * t_step.view(1, -1)
            epsilon_cat_end = torch.cat([epsilon, torch.ones(epsilon.shape) * self.epsilon_end], dim=0)
            return torch.max(epsilon_cat_end, dim=0)[0]
        else:
            raise TypeError(f"The variable ``t_step`` must be int or torch.Tensor, but got {type(t_step)}")




[docs]class ExpEpsilonDecay(EpsilonDecayBase):
    """
    The class for the exponential epsilon decay strategy.

    Args:
        epsilon_start (float):
            The starting value of the epsilon.
        epsilon_end (float):
            The ending value of the epsilon.
        time_length (int):
            The time length of the epsilon decaying from ``epsilon_start`` to ``epsilon_end``.
    """
    def __init__(self, epsilon_start: float, epsilon_end: float, time_length: int):
        super(ExpEpsilonDecay, self).__init__(epsilon_start, epsilon_end, time_length)

        self.decay_type = 'exp'
        self.exp_scaling = (-1) * self.time_length / np.log(self.epsilon_end) if self.epsilon_end > 0 else 1


[docs]    def getEpsilon(self, t_step: Union[int, torch.Tensor]) -> torch.Tensor:
        """
        Obtain the epsilon for the given t step.

        Args:
            t_step (int or torch.Tensor):
                The t step or steps.

        Returns:
            torch.Tensor: **epsilon**: The epsilon for the given ``t_step(int)`` with shape [1, ] or ``t_step(torch.Tensor)`` with shape [batch_size, ].
        """
        if isinstance(t_step, int):
            epsilon = min(self.epsilon_start, max(self.epsilon_end, np.exp(- t_step / self.exp_scaling)))
            return torch.as_tensor([epsilon])
        elif isinstance(t_step, torch.Tensor):
            epsilon = torch.exp(- t_step.view(1, -1) / self.exp_scaling)
            epsilon_cat_end = torch.cat([epsilon, torch.ones(epsilon.shape) * self.epsilon_end], dim=0)
            epsilon = torch.max(epsilon_cat_end, dim=0)[0].unsqueeze(0)
            epsilon_cat_start = torch.cat([epsilon, torch.ones(epsilon.shape) * self.epsilon_start], dim=0)
            return torch.min(epsilon_cat_start, dim=0)[0]
        else:
            raise TypeError(f"The variable ``t_step`` must be int or torch.Tensor, but got {type(t_step)}")

    """
    The base class for Action Selector.

    Args:
        epsilon_decay_fn (function):
            The function to create an epsilon_decay instance (return ``EpsilonDecayBase``).
        *epsilon_args:
            Arguments for creating an epsilon_decay instance.

        **epsilon_kwargs:
            Keyword arguments for creating an epsilon_decay instance.
    """
    def __init__(self, epsilon_decay_fn: Callable[..., EpsilonDecayBase], *epsilon_args, **epsilon_kwargs):
        self.epsilon_decay = epsilon_decay_fn(*epsilon_args, **epsilon_kwargs)
        self._epsilon = self.epsilon_decay.getEpsilon(0)

    @property
    def epsilon(self) -> float:
        """
        Obtain the current epsilon of the action_selector.

        Returns:
            float: **epsilon**: The epsilon of the action_selector.
        """
        return self._epsilon

[docs]    def select_actions(self, policy_outputs: torch.Tensor, avail_actions: torch.Tensor, t_step: Union[int, torch.Tensor], test_mode=False) -> torch.Tensor:
        """
        Select the actions based on the epsilon-greedy exploration strategy.

        Args:
            policy_outputs (torch.Tensor):
                The outputs of the policy, usually be the probability of the actions.
            avail_actions (torch.Tensor):
                The available actions at the current state.
            t_step (int or torch.Tensor):
                The t steps of the environments. When it is torch.Tensor, the shape should be [batch_size, 1]
            test_mode (bool, optional):
                Whether execute the test mode strategy, defaults to ``False``.

        Returns:
            torch.Tensor: **actions**: The actions with shape [batch_size, 1]
        """
        #t_step = 50000
        

        batch_size = len(t_step)
        if not test_mode:
            self._epsilon = self.epsilon_decay.getEpsilon(t_step) 
        else:
            self._epsilon = torch.zeros([batch_size])         # greedy action selection when it's under test condition

        # self._epsilon = self._epsilon.unsqueeze(-1)

        masked_q_values = policy_outputs.clone()
        masked_q_values[avail_actions == 0] = - float("inf")        # never select unavailable actions

        random_actions = torch.distributions.Categorical(avail_actions.float()).sample().long()
        pick_random_actions = (torch.rand(masked_q_values.shape[0]) < self._epsilon).long().unsqueeze(-1)

        actions = pick_random_actions * random_actions + (1 - pick_random_actions) * masked_q_values.max(dim=2)[1]

        return actions.reshape(batch_size, -1, 1)

[docs]    def select_actions(self, policy_outputs: torch.Tensor, avail_actions: torch.Tensor, t_step: Union[int, torch.Tensor], test_mode=False) -> torch.Tensor:
        masked_policies = policy_outputs.clone()
        masked_policies[avail_actions == 0] = 0.0

        if test_mode and self.test_greedy:
            actions = masked_policies.max(dim=1)[1]
        else:
            actions = torch.distributions.Categorical(masked_policies).sample().long()

        return actions
